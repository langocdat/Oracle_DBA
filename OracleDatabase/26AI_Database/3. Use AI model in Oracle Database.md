# Trường hợp 1: Import dữ liệu là file docx, pdf
  - Tạo bảng lưu trữ dữ liệu
  - Tự động Cắt nhỏ (Chunk) và Vector hóa tài liệu
    + Sử dụng hàm DBMS_VECTOR_CHAIN để bóc tách dữ liệu từ file word/pdf -> cắt nhỏ thành các đoạn văn ngắn -> Gọi model AI AIDEMO_DOC_MODEL để tạo Vector lưu vào bảng đích

drop table if exists AITEST_DOCUMENTATION_TAB purge;

## Step 2: Create table to store unstructured data chunks and associated vector embeddings
SQL> 
```
drop table if exists AITEST_DOCUMENTATION_CHUNKS purge;

CREATE TABLE AITEST_DOCUMENTATION_CHUNKS
   (
     doc_id          NUMBER,
     chunk_id        NUMBER,
     chunk_data      VARCHAR2(4000),
     chunk_embedding VECTOR
   );
```

## Step 3: Chuyển dữ liệu thành Vector (Embeddings) từ bảng AITEST_DOCUMENTATION_TAB lưu vào bảng AITEST_DOCUMENTATION_CHUNKS
SQL>
```
INSERT INTO AITEST_DOCUMENTATION_CHUNKS
 SELECT dt.id                      doc_id,
        et.embed_id                chunk_id,
        et.embed_data              chunk_data,
        To_vector(et.embed_vector) chunk_embedding
 FROM   AITEST_DOCUMENTATION_TAB dt,
        dbms_vector_chain.utl_to_embeddings( dbms_vector_chain.utl_to_chunks(dbms_vector_chain.utl_to_text(dt.data), json('{"normalize":"all", "max":300, "overlap":30}')),
        json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')) t,
        json_table(t.column_value, '$[*]' COLUMNS (embed_id NUMBER path '$.embed_id', embed_data VARCHAR2(4000) path '$.embed_data', embed_vector clob  
        path '$.embed_vector')) et;

 commit;
```
- Kiểm tra dữ liệu sau khi nạp của bảng AITEST_DOCUMENTATION_CHUNKS
  
SQL>
```
SELECT DOC_ID, COUNT(*) as total_chunks FROM AITEST_DOCUMENTATION_CHUNKS GROUP BY DOC_ID;
```
## Step 4: Thực hiện select 

- Tạo Query

SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
VARIABLE my_query_vec VECTOR;
DECLARE
    v_search_text VARCHAR2(1000) := 'Biên soạn'; 
BEGIN
    SELECT vector_embedding(AIDEMO_DOC_MODEL USING v_search_text AS data) INTO :my_query_vec;
END;
/
```

- Select
```
SET LINESIZE 150;
SET PAGESIZE 50;
COLUMN chunk_data FORMAT A100 WORD_WRAPPED;

-- Chạy lại câu lệnh SELECT để xem thành quả
SELECT doc_id, chunk_id, chunk_data FROM AITEST_DOCUMENTATION_CHUNKS 
ORDER BY vector_distance(chunk_embedding, :my_query_vec, COSINE) FETCH FIRST 1 ROWS ONLY;
```
# Trường hợp 2: Sử dụng bảng dữ liệu có sẵn
## Giải pháp 1: Add thêm 2 column mới
  - Column 1: cloumn dạng văn bản chứa nội dung tổng hợp của cả row
  - Column 2: chứa vector
### Step 1: Tạo bảng Employee
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
CREATE TABLE EMPLOYEE (
    ID         NUMBER PRIMARY KEY,
    TEN        VARCHAR2(100) NOT NULL,
    SDT        VARCHAR2(15),
    PHONG_BAN  VARCHAR2(50),
    LUONG      NUMBER(12,2)
);

INSERT INTO EMPLOYEE (ID, TEN, SDT, PHONG_BAN, LUONG)
SELECT
    LEVEL AS ID,
    'NhanVien_' || LEVEL AS TEN,
    '09' || LPAD(LEVEL, 8, '0') AS SDT,
    CASE MOD(LEVEL, 5)
        WHEN 0 THEN 'IT'
        WHEN 1 THEN 'HR'
        WHEN 2 THEN 'Finance'
        WHEN 3 THEN 'Marketing'
        ELSE 'Sales'
    END AS PHONG_BAN,
    8000000 + (LEVEL * 100000) AS LUONG
FROM DUAL
CONNECT BY LEVEL <= 100;

commit;
```

### Step 2: Add column
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
ALTER TABLE EMPLOYEE ADD (
    EMP_CONTEXT   VARCHAR2(4000),     
    EMP_EMBEDDING VECTOR
);

UPDATE EMPLOYEE 
SET EMP_CONTEXT = 'Nhân viên tên là ' || TEN || 
                  ', làm việc tại phòng ' || PHONG_BAN || 
                  ', số điện thoại liên hệ là ' || NVL(SDT, 'chưa cập nhật') || 
                  ' và đang có mức lương là ' || TO_CHAR(LUONG, '999,999,999') || ' đồng.';
COMMIT;

UPDATE EMPLOYEE
SET EMP_EMBEDDING = dbms_vector_chain.utl_to_embedding(
    EMP_CONTEXT, 
    json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')
);
COMMIT;
```
### step 3: SELECT 
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
SELECT ID, TEN, PHONG_BAN, LUONG, EMP_CONTEXT
FROM EMPLOYEE
ORDER BY VECTOR_DISTANCE(
    EMP_EMBEDDING, 
    dbms_vector_chain.utl_to_embedding('Nhân viên_100', json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')), 
    COSINE
)
FETCH FIRST 1 ROWS ONLY;
```

## Giải pháp 2: Tạo bảng vector song song
### Step 1: Tạo bảng lưu vector
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
CREATE TABLE EMPLOYEE_VECTORS (
    ID            NUMBER PRIMARY KEY,
    EMP_CONTEXT   VARCHAR2(4000),
    EMP_EMBEDDING VECTOR(384, FLOAT32),
    CONSTRAINT fk_emp_vector FOREIGN KEY (ID) REFERENCES EMPLOYEE(ID) ON DELETE CASCADE
);
```
### Step 2: Đồng bộ dữ liệu và sinh Vector vào bảng mới
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
INSERT INTO EMPLOYEE_VECTORS (ID, EMP_CONTEXT, EMP_EMBEDDING)
SELECT 
    ID,
    'Nhân viên tên là ' || TEN || ', làm việc tại phòng ' || PHONG_BAN || '.',
    dbms_vector_chain.utl_to_embedding(
        'Nhân viên tên là ' || TEN || ', làm việc tại phòng ' || PHONG_BAN || '.', 
        json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')
    )
FROM EMPLOYEE;
COMMIT;
```
### Step 3: Test select
SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
SELECT e.ID, e.TEN, e.PHONG_BAN, v.EMP_CONTEXT
FROM EMPLOYEE e

    v.EMP_EMBEDDING, 
    dbms_vector_chain.utl_to_embedding('Ai làm kế toán?', json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')), 
    COSINE
)
FETCH FIRST 2 ROWS ONLY;
```
