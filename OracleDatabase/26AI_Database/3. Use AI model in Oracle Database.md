## Step 2: Create table to store unstructured data chunks and associated vector embeddings
SQL> 
```
drop table if exists AITEST_DOCUMENTATION_CHUNKS purge;

CREATE TABLE AITEST_DOCUMENTATION_CHUNKS
   (
     doc_id          NUMBER,
     chunk_id        NUMBER,
     chunk_data      VARCHAR2(4000),
     chunk_embedding VECTOR
   );
```

## Step 3: Chuyển dữ liệu thành Vector (Embeddings) từ bảng AITEST_DOCUMENTATION_TAB lưu vào bảng AITEST_DOCUMENTATION_CHUNKS
SQL>
```
INSERT INTO AITEST_DOCUMENTATION_CHUNKS
 SELECT dt.id                      doc_id,
        et.embed_id                chunk_id,
        et.embed_data              chunk_data,
        To_vector(et.embed_vector) chunk_embedding
 FROM   AITEST_DOCUMENTATION_TAB dt,
        dbms_vector_chain.utl_to_embeddings( dbms_vector_chain.utl_to_chunks(dbms_vector_chain.utl_to_text(dt.data), json('{"normalize":"all", "max":300, "overlap":30}')),
        json('{"provider":"database", "model":"AIDEMO_DOC_MODEL"}')) t,
        json_table(t.column_value, '$[*]' COLUMNS (embed_id NUMBER path '$.embed_id', embed_data VARCHAR2(4000) path '$.embed_data', embed_vector clob  
        path '$.embed_vector')) et;

 commit;
```
- Kiểm tra dữ liệu sau khi nạp của bảng AITEST_DOCUMENTATION_CHUNKS
  
SQL>
```
SELECT DOC_ID, COUNT(*) as total_chunks FROM AITEST_DOCUMENTATION_CHUNKS GROUP BY DOC_ID;
```
## Step 4: Thực hiện select 

- Tạo Query

SQL> sqlplus AITEST/welcome1@//localhost:1521/ORCLPDB1
```
VARIABLE my_query_vec VECTOR;
DECLARE
    v_search_text VARCHAR2(1000) := 'Biên soạn'; 
BEGIN
    SELECT vector_embedding(AIDEMO_DOC_MODEL USING v_search_text AS data) INTO :my_query_vec;
END;
/
```

- Select
```
SET LINESIZE 150;
SET PAGESIZE 50;
COLUMN chunk_data FORMAT A100 WORD_WRAPPED;

-- Chạy lại câu lệnh SELECT để xem thành quả
SELECT doc_id, chunk_id, chunk_data FROM AITEST_DOCUMENTATION_CHUNKS 
ORDER BY vector_distance(chunk_embedding, :my_query_vec, COSINE) FETCH FIRST 1 ROWS ONLY;
```
