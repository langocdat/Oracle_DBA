- Change attribute action
```
BEGIN
    DBMS_SCHEDULER.SET_ATTRIBUTE (
        name      => 'JOB_REFRESH_PASSENGER_TRAFFIC_MV',
        attribute => 'JOB_ACTION',
        value     => 'BEGIN DBMS_MVIEW.REFRESH(''PASSENGER_TRAFFIC_MV'', ''C''); END;'
    );
END;
/
```

- Change repeat interval attribute
```
BEGIN
    DBMS_SCHEDULER.SET_ATTRIBUTE (
        name      => 'JOB_REFRESH_PASSENGER_TRAFFIC_MV',
        attribute => 'REPEAT_INTERVAL',
        value     => 'FREQ=MINUTELY; INTERVAL=15'
    );
END;
/
```

- Check over information
```
ALTER SESSION SET NLS_DATE_FORMAT = 'DD-MON-YYYY HH24:MI:SS';
set lines 200;
set pages 200;
col job_name format a30;
col job_type format a30;
col enabled format a20;
col repeat_interval format a20;
col last_start_date format a20;
col next_run_date format a20;
SELECT job_name,JOB_ACTION, enabled, state, repeat_interval, last_start_date, next_run_date
FROM dba_scheduler_jobs
WHERE job_name in ('JOB_REFRESH_BLACKLIST_PASSENGER','JOB_REFRESH_MV_BLACKLIST_SYNC','JOB_REFRESH_MV_PASSENGER_TRANS','JOB_REFRESH_MV_PASSENGER_TRANS_EGATE','JOB_REFRESH_MV_RCF_PASSENGER_TRANS','JOB_REFRESH_PASSENGER_TRAFFIC_MV');

SELECT job_name, status, run_duration
FROM dba_scheduler_job_run_details
WHERE job_name in ('JOB_REFRESH_BLACKLIST_PASSENGER','JOB_REFRESH_MV_BLACKLIST_SYNC','JOB_REFRESH_MV_PASSENGER_TRANS','JOB_REFRESH_MV_PASSENGER_TRANS_EGATE','JOB_REFRESH_MV_RCF_PASSENGER_TRANS','JOB_REFRESH_PASSENGER_TRAFFIC_MV') ORDER BY actual_start_date DESC;
```

- Stop Job -> Disable Job
```
BEGIN
  DBMS_SCHEDULER.STOP_JOB (
    job_name => 'IMS_2017.JOB_REFRESH_MV_RCF_PASSENGER_TRANS',
    force    => TRUE
  );
END;
/

BEGIN
    DBMS_SCHEDULER.DISABLE (
        name => 'JOB_REFRESH_MV_RCF_PASSENGER_TRANS'
    );
```
