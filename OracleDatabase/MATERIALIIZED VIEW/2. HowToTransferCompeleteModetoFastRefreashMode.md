# 3. Check thông tin của MVIEW
SQL
```
set lines 200;
col owner format a20;
col mview_name format a30;
col refresh_method format a20;
col refresh_mode format a20;
SELECT owner, mview_name, refresh_method, refresh_mode, last_refresh_date
FROM dba_mviews WHERE mview_name = 'BLACKLIST_PASSENGER' and owner = 'BPDT_NEW';
```

# 4. Check thông tin các JOB
SQL
```
SELECT job_name, enabled, state, repeat_interval, last_start_date, next_run_date
FROM dba_scheduler_jobs
WHERE job_name in ('JOB_REFRESH_BLACKLIST_PASSENGER','JOB_REFRESH_MV_BLACKLIST_SYNC','JOB_REFRESH_MV_PASSENGER_TRANS','JOB_REFRESH_MV_PASSENGER_TRANS_EGATE','JOB_REFRESH_MV_RCF_PASSENGER_TRANS','JOB_REFRESH_PASSENGER_TRAFFIC_MV');


SELECT job_name, status, run_duration
FROM dba_scheduler_job_run_details
WHERE job_name in ('JOB_REFRESH_BLACKLIST_PASSENGER','JOB_REFRESH_MV_BLACKLIST_SYNC','JOB_REFRESH_MV_PASSENGER_TRANS','JOB_REFRESH_MV_PASSENGER_TRANS_EGATE','JOB_REFRESH_MV_RCF_PASSENGER_TRANS','JOB_REFRESH_PASSENGER_TRAFFIC_MV') ORDER BY actual_start_date DESC;
```
