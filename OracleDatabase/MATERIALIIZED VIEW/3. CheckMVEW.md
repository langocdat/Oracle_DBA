ALTER SESSION SET NLS_DATE_FORMAT = 'DD-MON-YYYY HH24:MI:SS';
SELECT  m.owner, m.mview_name, m.last_refresh_date, r.next_date, r.interval FROM dba_mviews m JOIN dba_refresh_children r ON m.owner = r.owner AND m.mview_name = r.name
WHERE m.mview_name = 'MV_TEST' and m.owner='SYS';
